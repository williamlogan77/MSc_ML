# Create a function so that the same modification can be done twice
def SortData(identity, transaction):
    # Read in both data sets and store in the respective variable
    identity = pd.read_csv(identity)
    transaction = pd.read_csv(transaction)

    # To merge the two data sets into one data set using merge, it must be done
    # with the correct syntax. This statement checks for the biggest dataset and
    # ensures the correct syntax (even if the input is backwards).
    if identity.shape[1] > transaction.shape[1]:
        # Merge the datasets, by either adding the new "TransactionID"s, or
        # adding extra columns to those "TransactionID"s that already exist
        data = identity.merge(transaction, how="left", on="TransactionID")
    else:
        data = transaction.merge(identity, how="left", on="TransactionID")

    # Define a new boolean dataframe that contains information about columns
    # that contain 0 values
    cond = (data.isna().sum() / data.shape[0] != 1).values
    # Delete these columns (any column with no values in)
    data = data.loc[:, cond]

    # Import the imputer from SKLearn
    from sklearn.impute import SimpleImputer
    # Define the imputer class that will impute using the most frequent values
    # (the modal values)
    imputer = SimpleImputer(strategy="most_frequent", verbose=2)

    # Impute on the merged dataset and transform the data into the imputed data
    imputed = imputer.fit_transform(data)
    # Create a new dataframe containing all of the imputed data with the
    # respective colum names and indexes
    imputed_data = pd.DataFrame(
        imputed, columns=data.columns, index=data.index)

    # Loop through each column
    for i in imputed_data.keys():
        # Change the datatype of the column where possible, if not possible just
        # skip the column
        try:
            imputed_data[i] = imputed_data[i].astype("float64")
        except ValueError:
            continue

    # Import the encoder from sklearn
    from sklearn.preprocessing import LabelEncoder

    # Initialise the encoder class
    enc = LabelEncoder()
    # Select the columns that are not floats (i.e. non-numerical) and apply the
    # encoder to the data in the remaining columns
    encoded_data = imputed_data.select_dtypes(
        include="object").apply(enc.fit_transform)

    # Remove the previous non-numerical data from the data frame
    imputed_data.drop(columns=[x for x in imputed_data.select_dtypes(
        include="object")], inplace=True)
    # Put the encoded values back into the dataset
    imputed_data.join(encoded_data)

    # Return the dataframe after imputing, encoding and manipulating the
    # original data
    return imputed_data

# Define a function containing analytics
def Diagnostics(model, x_test=analysis_test_X, y_test=analysis_test_Y):
    # Make predictions based on the testing data
    model_predictions = model.predict(x_test)
    # Output the confusion matrix
    print(classification_report(y_test, model_predictions))

    # Initialise a figure object
    fig, ax = plt.subplots(figsize=(10, 10))
    # Plot the ROC curve using the probabilities
    plot = skplt.plot_roc(y_test, model.predict_proba(x_test), ax=ax)
    # Save the figure
    plt.savefig("./ROC_best_model.png")

    # Output the accuracy and AUC score
    print("Accuracy: %.4g" % metrics.accuracy_score(y_test, model_predictions))
    print("AUC Score (Train): %f" % metrics.roc_auc_score(
        y_test, model.predict_proba(x_test)[:, 1]))
    # Initialise a figure object
    fig, ax = plt.subplots(figsize=(15, 15))
    # Plot the
    xg.plot_importance(model, max_num_features=20, ax=ax)
    plt.tight_layout()
    plt.savefig("./output.png")
    plt.show()
